CREATE TABLE fakultas(
    id_fakultas BIGINT  PRIMARY KEY AUTO_INCREMENT,
    nama_fakultas VARCHAR (255)
    );

CREATE TABLE jurusan(
    id_jurusan BIGINT  PRIMARY KEY AUTO_INCREMENT,
    nama_jurusan VARCHAR (255),
    id_fakultas BIGINT, FOREIGN KEY(id_fakultas) REFERENCES fakultas(id_fakultas)
    );

CREATE TABLE mahasiswa(
    npm BIGINT  PRIMARY KEY AUTO_INCREMENT,
    nama VARCHAR (255),
    id_jurusan BIGINT, FOREIGN KEY(id_jurusan) REFERENCES jurusan(id_jurusan)
    );


CREATE TABLE kabupaten(
    id_kabupaten BIGINT  PRIMARY KEY AUTO_INCREMENT,
    nama_kabupaten VARCHAR (255)
    )

CREATE TABLE kecamatan(
    id_kecamatan BIGINT  PRIMARY KEY AUTO_INCREMENT,
    nama_kecamatan VARCHAR (255),
    id_kabupaten BIGINT, FOREIGN KEY(id_kabupaten) REFERENCES kabupaten(id_kabupaten)
    );

CREATE TABLE desa( 
	id_desa BIGINT PRIMARY KEY AUTO_INCREMENT, 
	nama_desa VARCHAR (255), 
	id_kecamatan BIGINT, FOREIGN KEY(id_kecamatan) REFERENCES kecamatan(id_kecamatan) 
	);

CREATE TABLE penempatan(
    id_penempatan BIGINT  PRIMARY KEY AUTO_INCREMENT,
	id_desa BIGINT, FOREIGN KEY(id_desa) REFERENCES desa(id_desa),
    npm BIGINT, FOREIGN KEY(npm) REFERENCES mahasiswa(npm)
    )

CREATE TABLE dpl(
    nip_dpl INT  PRIMARY KEY AUTO_INCREMENT,
	nama_dpl VARCHAR(255)
    )

CREATE TABLE periode(
    id_periode BIGINT PRIMARY KEY AUTO_INCREMENT,
    periode INT (10),
    tahun YEAR (4)
    )


CREATE TABLE penempatan_dpl(
    id_penempatan_dpl BIGINT PRIMARY KEY AUTO_INCREMENT,
    npm BIGINT, FOREIGN KEY(npm) REFERENCES mahasiswa(npm),
  	id_desa BIGINT, FOREIGN KEY(id_desa) REFERENCES desa(id_desa)
    )

CREATE TABLE lokasi_terdaftar(
    id_lokasi_terdaftar BIGINT PRIMARY KEY AUTO_INCREMENT,
  	id_desa BIGINT, FOREIGN KEY(id_desa) REFERENCES desa(id_desa),
    id_periode BIGINT, FOREIGN KEY(id_periode) REFERENCES periode(id_periode)
    )
